    # Load the data
    X, y = data_10_classes(return_X_y=True, as_frame=False)

    # Create an instance of the LogisticRegression
    model = LogisticRegression(max_iter=int(1e5), random_state=0)

    # Plot!
    ax = skplt.estimators.plot_learning_curve(
